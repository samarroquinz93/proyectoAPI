from flask import Flask, render_template_string, request, redirect app = Flask(__name__) reportes = [] html = """ Comunidad Reporta

📍 Comunidad Reporta

{% for r in reportes %}

🚨 {{r.titulo}}

{{r.descripcion}}

📍 Ubicación: {{r.ubicacion}}

Estado: {{r.estado}}

{% endfor %}
""" @app.route("/") def inicio(): return render_template_string(html, reportes=reportes) @app.route("/crear", methods=["POST"]) def crear(): reporte = { "titulo": request.form["titulo"], "descripcion": request.form["descripcion"], "ubicacion": request.form["ubicacion"], "estado": request.form["estado"] } reportes.append(reporte) return redirect("/") app.run(debug=True)
